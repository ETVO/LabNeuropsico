DROP TABLE IF EXISTS user;

CREATE TABLE user (
    id_user BIGINT PRIMARY KEY,
    login VARCHAR(15) UNIQUE NOT NULL,
    senha TEXT NOT NULL,
    email VARCHAR(254) UNIQUE NOT NULL,
    nome TEXT NOT NULL,
    setor CHAR NOT NULL,
    obs TEXT,
    excluido BOOLEAN NOT NULL DEFAULT false
);
